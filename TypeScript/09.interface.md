# interface

```js
interface Person2 {
  name: string;
  // age가 있어도 되고 없어도 될 때 물음표를 붙여준다.
  age?: number;
}

function hello2(person: Person2): void {
  console.log(`안녕하세요! ${person.name}입니다.`);
}


hello2({ name: "hail", age: 27 });
hello2({ name: "jeong" });
```

```js
interface Person3 {
  name: string;
  age?: number;
  [index: string]: any;
}

function hello3(person: Person3): void {
  console.log(`안녕하세요 ${person.name} 입니다.`);
}

const p31: Person3 = {
  name: "hail",
  age: 27,
};

const p32: Person3 = {
  name: "jeong",
  systers: ["zhang", "zheng"],
};

const p33: Person3 = {
  name: "Kim",
  father: p31,
  mother: p32,
};

hello3(p33);
```

```js
interface Person4 {
  name: string;
  age: number;
  hello(): void;
}

const p41: Person4 = {
  name: "hail",
  age: 27,
  hello: function (): void {
    console.log(`안녕하세요 제 이름은 ${this.name} 입니다.`);
  },
};

const p42: Person4 = {
  name: "hail",
  age: 27,
  hello(): void {
    console.log(`안녕하세요 제 이름은 ${this.name} 입니다.`);
  },
};

// 화살표 함수에서는 this를 사용할 수 없다.
const p43: Person4 = {
  name: "hail",
  age: 27,
  hello: (): void => {
    console.log(`안녕하세요 제 이름은 ${this.name} 입니다.`);
  },
};

// npx tsc로 컴파일 한 후에 node interface4.js 로 실행시키면 결과 값으로 안녕하세요 제 이름은 hail 입니다. 가 작성된다.
p41.hello();
p42.hello();
```
