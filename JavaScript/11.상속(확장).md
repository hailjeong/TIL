
```javascript
class hail {
  constructor(name, wheel) {
    this.name = name;
    this.wheel = wheel;
  }
}

class Bicycle extends hail {
  constructor(name, wheel) {
    super(name, wheel);
  }
}

class Car extends hail {
  constructor(name, wheel, license) {
    super(name, wheel);
    this.license = license;
  }
}
```
